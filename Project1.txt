def turn_right():
    turn_left()
    turn_left()
    turn_left()

while at_goal() == False:
    if is_facing_north():
        while front_is_clear():
            move()
    if wall_on_right() != True:
        turn_right()
        move()
    elif wall_on_right() == True and front_is_clear():
        move()
    elif wall_on_right() == True and not front_is_clear():
        turn_left()
